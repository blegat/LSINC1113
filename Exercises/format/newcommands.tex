% Fomatting titles for sessions associated with a counter
\newcounter{session}
\newcommand{\session}[1]{{
    \newpage
    \begin{center}
        \sf \Large \centering
        \maintitle{LSINC1113 - Compléments de mathématiques}\\[.8em]
        {\LARGE \textbf{#1}}\\
        \hrulefill
    \end{center}
    \vspace*{0.5cm}
}}

% Formatting equations
\counterwithin*{equation}{session}
\counterwithin*{figure}{session}
\counterwithin*{footnote}{session}

% Formatting exercices/solutions according to the counter session
\theoremstyle{plain}
    \newtheorem{theorem}{Théorème}
    \counterwithin*{theorem}{session} 
\theoremstyle{definition}
    \newtheorem{exercise}{Exercice}
    \counterwithin*{exercise}{session} 
\newenvironment{note}{
    \small {\it Note.} \hskip1ex
    }
    {}

\fi
